// sort first 2 elements 
...
// sort all n items 

// *** Pseudo Code ***

for i = 2 .. n
  // we will be sorting A[1:i]
  // we will compare an element of the array A, in A[1: i - 1] starting from index ..
  j = i - 1
  // .. with element 
  Ai = A[i]
  // we don't know how many comparisons to make so we use a while loop: don't forget to check edge case and go outside of ranage
  while j > 0 and Ai < A[j]
    // right shift 
    A[j + 1] = A[j]
    // change the index so we don't create 1 infinite loop 
    j = j - 1
 // when we exit the loop this mean Ai => A[j] or we've reached the front of the array, so 
 A[j + 1] = Ai
 
// running this procedure results in the sorted array A.
